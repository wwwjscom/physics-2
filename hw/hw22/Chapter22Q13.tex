		F = N_{A}e
	\end{equation}
	
	\begin{center}
		$N_{A}$ represents Avogadro's constant	
	\end{center}
	
	\begin{equation}
		e = 1.602*10^{-19}C
	\end{equation}
	
	\begin{center}
		Thus
	\end{center}
	
	\begin{equation}
		F = 9.6487*10^{4} \frac{C}{mol}
	\end{equation}
